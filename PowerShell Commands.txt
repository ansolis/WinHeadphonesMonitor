Get headphones battery:

function Get-HeadsetBattery {
    $device = Get-PnpDevice | Where-Object { $_.FriendlyName -eq "WH-1000XM3 Hands-Free AG" } -ErrorAction SilentlyContinue
    
    if ($device) {
        $battery = Get-PnpDeviceProperty -InstanceId $device.InstanceId -KeyName '{104EA319-6EE2-4701-BD47-8DDBF425BBE5} 2' -ErrorAction SilentlyContinue
        if ($null -ne $battery.Data) {
            Write-Host "Sony WH-1000XM3 Battery: " -NoNewline
            Write-Host "$($battery.Data)%" -ForegroundColor Green
        } else {
            Write-Host "Battery level not reported. Is the headset connected?" -ForegroundColor Yellow
        }
    } else {
        Write-Host "Headset device not found." -ForegroundColor Red
    }
}
